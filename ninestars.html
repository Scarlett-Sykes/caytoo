function openBrandModal(brandId) { const signal = signals.find(s => s.BrandID === brandId); if (!signal) return; const form = document.getElementById('brandForm'); form.brandId.value = signal.BrandID || ''; form.brandName.value = signal.BrandName || ''; form.category.value = signal.Category || ''; form.subcategory.value = signal.Subcategory || ''; form.city.value = signal.City || ''; form.region.value = signal.Region || ''; form.country.value = signal.Country || ''; form.revenue.value = signal.Revenue || ''; document.getElementById('brandModal').style.display = "block"; }
